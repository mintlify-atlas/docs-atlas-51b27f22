Group chats enable multi-party conversations with advanced features like member management, permissions, and admin controls.

## Creating Groups

### Create a Basic Group

Create a group with default settings:

```swift Conversations.swift:268-298

```



## Best Practices

<AccordionGroup>
  <Accordion title="Sync before operations">
    Always sync the group before performing operations:
    
    ```swift
    try await group.sync()
    
    // Now safe to access current state
    let members = try await group.members
    let isAdmin = try group.isAdmin(inboxId: myInboxId)
    ```
  </Accordion>

  <Accordion title="Handle permission errors">
    Check permissions before attempting operations:
    
    ```swift
    if try group.isAdmin(inboxId: client.inboxID) {
        try await group.addMembers(inboxIds: [newMember])
    } else {
        print("Not authorized to add members")
    }
    ```
  </Accordion>

  <Accordion title="Use admin-only for sensitive groups">
    For sensitive groups, use admin-only permissions:
    
    ```swift
    let group = try await client.conversations.newGroup(
        with: memberIds,
        permissions: .adminOnly
    )
    ```
  </Accordion>

  <Accordion title="Provide group context">
    Set meaningful names and descriptions:
    
    ```swift
    let group = try await client.conversations.newGroup(
        with: members,
        groupName: "Engineering Team",
        groupDescription: "Daily standup and project coordination",
        groupImageUrlSquare: teamLogoUrl
    )
    ```
  </Accordion>

  <Accordion title="Handle membership changes">
    Listen for membership update messages:
    
    ```swift
    Client.register(codec: GroupUpdatedCodec())
    
    let messages = try await group.messages()
    
    for message in messages {
        if message.contentTypeId == ContentTypeGroupUpdated {
            let update: GroupUpdated = try message.content()
            print("Added: \(update.addedInboxes)")
            print("Removed: \(update.removedInboxes)")
        }
    }
    ```
  </Accordion>
</AccordionGroup>

## Examples

### Create Team Workspace

```swift
func createTeamWorkspace(
    name: String,
    members: [String],
    admins: [String]
) async throws -> Group {
    // Create group with admin-only permissions
    let group = try await client.conversations.newGroup(
        with: members,
        permissions: .adminOnly,
        groupName: name,
        groupDescription: "Team collaboration space"
    )
    
    // Promote specified users to admin
    for adminInboxId in admins {
        try await group.addAdmin(inboxId: adminInboxId)
    }
    
    // Enable disappearing messages (7 days)
    let settings = DisappearingMessageSettings(
        disappearStartingAtNs: Date().millisecondsSinceEpoch,
        retentionDurationInNs: 7 * 24 * 60 * 60 * 1_000_000_000
    )
    try await group.updateDisappearingMessageSettings(settings)
    
    return group
}
```

### Monitor Group Activity

```swift
class GroupMonitor {
    let group: Group
    private var streamTask: Task<Void, Never>?
    
    init(group: Group) {
        self.group = group
    }
    
    func startMonitoring() {
        streamTask = Task {
            let stream = group.streamMessages()
            
            for try await message in stream {
                await handleMessage(message)
            }
        }
    }
    
    func handleMessage(_ message: DecodedMessage) async {
        // Handle different message types
        switch message.contentTypeId {
        case ContentTypeText:
            let text: String = try! message.content()
            print("Message: \(text)")
            
        case ContentTypeGroupUpdated:
            try await group.sync()
            let members = try await group.members
            print("Members updated, count: \(members.count)")
            
        default:
            print("Other message type")
        }
    }
    
    func stopMonitoring() {
        group.endStream()
        streamTask?.cancel()
    }
}
```

### Batch Member Management

```swift
func updateGroupMembers(
    group: Group,
    toAdd: [String],
    toRemove: [String],
    toPromote: [String]
) async throws {
    // Remove members first
    if !toRemove.isEmpty {
        try await group.removeMembers(inboxIds: toRemove)
        print("Removed \(toRemove.count) members")
    }
    
    // Add new members
    if !toAdd.isEmpty {
        let result = try await group.addMembers(inboxIds: toAdd)
        print("Added \(result.addedMembers.count) members")
        
        if !result.failedMembers.isEmpty {
            print("Failed to add: \(result.failedMembers)")
        }
    }
    
    // Promote to admin
    for inboxId in toPromote {
        try await group.addAdmin(inboxId: inboxId)
    }
    
    // Sync to get latest state
    try await group.sync()
    
    let finalMembers = try await group.members
    print("Final member count: \(finalMembers.count)")
}
```
