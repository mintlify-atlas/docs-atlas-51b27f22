The XMTP iOS SDK provides powerful methods for retrieving message history and listening for new messages in real-time.

## Fetching Message History

### Get All Messages

Retrieve all messages from a conversation:

```swift Conversation.swift:302-336
let conversation = try await client.conversations.list().first!

// Fetch all messages
let messages = try await conversation.messages()

for message in messages {
    print("\(message.senderInboxId): \(message.content())")
}
```

### Get Recent Messages

Limit the number of messages returned:

```swift Group.swift:550-623
let recentMessages = try await conversation.messages(
    limit: 50,
    direction: .descending  // Most recent first
)
```

### Get Messages with Filters

Apply filters to retrieve specific messages:

```swift Group.swift:550-623
let now = Date().millisecondsSinceEpoch
let oneDayAgo = now - (24 * 60 * 60 * 1000 * 1_000_000)

let messages = try await conversation.messages(
    beforeNs: now,
    afterNs: oneDayAgo,
    limit: 100,
    direction: .descending,
    deliveryStatus: .published  // or .unpublished, .failed, .all
)
```

## Understanding Message Types

### Raw Messages vs. Enriched Messages

XMTP provides two ways to retrieve messages:

<Tabs>
  <Tab title="Raw Messages (messages)">
    Returns messages without enriched metadata. Reactions, replies, and other content types are separate messages:

    ```swift Conversation.swift:291-336
    let messages = try await conversation.messages(limit: 50)
    
    // Reactions and replies are separate messages
    for message in messages {
        if message.contentTypeId == ContentTypeReaction {
            let reaction: Reaction = try message.content()
            print("Reaction: \(reaction.content)")
        } else if message.contentTypeId == ContentTypeText {
            let text: String = try message.content()
            print("Text: \(text)")
        }
    }
    ```
  </Tab>

  <Tab title="Enriched Messages (enrichedMessages)">
    **Recommended for UI rendering.** Returns messages with reactions, replies, and metadata "baked in":

    ```swift Conversation.swift:393-442
    let messages = try await conversation.enrichedMessages(limit: 50)
    
    for message in messages {
        let text: String = try message.content()
        
        // Access enriched metadata
        if let reactions = message.reactions {
            print("Reactions: \(reactions.count)")
        }
        
        if let reply = message.reply {
            print("Reply to: \(reply.reference)")
        }
    }
    ```
  </Tab>
</Tabs>

### When to Use Each Method

- **Use `messages()`**: When you need raw message data or are building custom message processing
- **Use `enrichedMessages()`**: When building UI that displays messages with reactions, replies, and other metadata

## Streaming New Messages

Listen for new messages in real-time:

```swift Group.swift:489-521
let conversation = try await client.conversations.list().first!

let stream = conversation.streamMessages()

for try await message in stream {
    print("New message from \(message.senderInboxId): \(message.content())")
}
```

### Stream with Cleanup

Handle stream closure:

```swift Group.swift:489-521
let stream = conversation.streamMessages {
    print("Stream closed")
    // Perform cleanup
}

for try await message in stream {
    print("New message: \(message.content())")
}
```

### Stop Streaming

End the message stream:

```swift Group.swift:485-487
let group = try await client.conversations.listGroups().first!

let streamTask = Task {
    for try await message in group.streamMessages() {
        print("Message: \(message.content())")
    }
}

// Later, stop streaming
group.endStream()
streamTask.cancel()
```

## Processing Different Content Types

### Generic Content Decoding

```swift
func processMessage(_ message: DecodedMessage) {
    switch message.contentTypeId {
    case ContentTypeText:
        let text: String = try! message.content()
        displayText(text)
        
    case ContentTypeAttachment:
        let attachment: Attachment = try! message.content()
        displayAttachment(attachment)
        
    case ContentTypeReaction:
        let reaction: Reaction = try! message.content()
        displayReaction(reaction)
        
    case ContentTypeReply:
        let reply: Reply = try! message.content()
        displayReply(reply)
        
    case ContentTypeReadReceipt:
        updateReadStatus()
        
    default:
        print("Unknown content type: \(message.contentTypeId)")
    }
}
```

## Pagination

### Load More Messages

```swift
var allMessages: [DecodedMessage] = []
var oldestTimestamp: Int64? = nil
let pageSize = 50

repeat {
    let page = try await conversation.messages(
        beforeNs: oldestTimestamp,
        limit: pageSize,
        direction: .descending
    )
    
    allMessages.append(contentsOf: page)
    oldestTimestamp = page.last?.sentAtNs
    
    print("Loaded \(page.count) messages, total: \(allMessages.count)")
    
} while oldestTimestamp != nil && !page.isEmpty
```

### Load Newer Messages

```swift
let newestTimestamp = messages.first?.sentAtNs ?? 0

let newerMessages = try await conversation.messages(
    afterNs: newestTimestamp,
    direction: .ascending
)

if !newerMessages.isEmpty {
    print("Found \(newerMessages.count) new messages")
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Use enrichedMessages for UI">
    When building chat interfaces, prefer `enrichedMessages()` for better performance:
    
    ```swift
    // Better for UI - reactions and replies are included
    let messages = try await conversation.enrichedMessages(limit: 50)
    
    for message in messages {
        let text: String = try message.content()
        let reactions = message.reactions ?? []
        
        displayMessage(text: text, reactions: reactions)
    }
    ```
  </Accordion>

  <Accordion title="Sync before fetching">
    Always sync the conversation before fetching messages:
    
    ```swift
    try await conversation.sync()
    let messages = try await conversation.messages()
    ```
  </Accordion>

  <Accordion title="Handle content types gracefully">
    Always check content types before decoding:
    
    ```swift
    for message in messages {
        switch message.contentTypeId {
        case ContentTypeText:
            let text: String = try message.content()
            print(text)
        default:
            // Handle unknown types
            if let fallback = message.fallback {
                print("Unsupported: \(fallback)")
            }
        }
    }
    ```
  </Accordion>

  <Accordion title="Implement pagination">
    Load messages in pages for better performance:
    
    ```swift
    func loadMoreMessages() async throws {
        let lastMessage = displayedMessages.last
        
        let more = try await conversation.messages(
            beforeNs: lastMessage?.sentAtNs,
            limit: 50
        )
        
        displayedMessages.append(contentsOf: more)
    }
    ```
  </Accordion>

  <Accordion title="Cache message data">
    Store messages locally to reduce network requests:
    
    ```swift
    // First load from cache
    var messages = database.getMessages(conversationId: conversation.id)
    
    // Then sync and update
    try await conversation.sync()
    let newMessages = try await conversation.messages(
        insertedAfterNs: lastSyncTime
    )
    
    if !newMessages.isEmpty {
        database.saveMessages(newMessages)
        messages.append(contentsOf: newMessages)
    }
    ```
  </Accordion>

  <Accordion title="Use streams for real-time updates">
    Stream messages for real-time chat experiences:
    
    ```swift
    Task {
        let stream = conversation.streamMessages()
        
        for try await message in stream {
            await MainActor.run {
                displayNewMessage(message)
                playNotificationSound()
            }
        }
    }
    ```
  </Accordion>
</AccordionGroup>

## Examples

### Build a Message List

```swift
class MessageListViewModel: ObservableObject {
    @Published var messages: [DecodedMessageV2] = []
    private var streamTask: Task<Void, Never>?
    
    func loadMessages(conversation: Conversation) async {
        do {
            // Sync first
            try await conversation.sync()
            
            // Load initial messages
            messages = try await conversation.enrichedMessages(
                limit: 50,
                direction: .descending
            )
            
            // Start streaming new messages
            startStreaming(conversation)
            
        } catch {
            print("Failed to load messages: \(error)")
        }
    }
    
    func startStreaming(_ conversation: Conversation) {
        streamTask = Task { [weak self] in
            let stream = conversation.streamMessages()
            
            for try await message in stream {
                guard !Task.isCancelled else { break }
                
                await MainActor.run {
                    self?.messages.insert(
                        DecodedMessageV2.create(ffiMessage: message),
                        at: 0
                    )
                }
            }
        }
    }
    
    func loadMore() async {
        guard let oldest = messages.last else { return }
        
        let more = try? await conversation.enrichedMessages(
            beforeNs: oldest.sentAtNs,
            limit: 50
        )
        
        if let more = more {
            messages.append(contentsOf: more)
        }
    }
    
    deinit {
        streamTask?.cancel()
    }
}
```

### Download Remote Attachment

```swift
func downloadRemoteAttachment(
    from message: DecodedMessage
) async throws -> Attachment {
    guard message.contentTypeId == ContentTypeRemoteAttachment else {
        throw AttachmentError.wrongType
    }
    
    // Get remote attachment
    let remote: RemoteAttachment = try message.content()
    
    print("Downloading from: \(remote.url)")
    
    // Fetch encrypted content
    let encodedContent = try await remote.content()
    
    // Decode attachment
    let codec = AttachmentCodec()
    let attachment = try codec.decode(content: encodedContent)
    
    print("Downloaded: \(attachment.filename)")
    return attachment
}
```

### Group Messages by Sender

```swift
struct GroupedMessages {
    let sender: String
    let messages: [DecodedMessage]
}

func groupMessagesBySender(
    _ messages: [DecodedMessage]
) -> [GroupedMessages] {
    let grouped = Dictionary(grouping: messages) { $0.senderInboxId }
    
    return grouped.map { sender, messages in
        GroupedMessages(
            sender: sender,
            messages: messages.sorted { $0.sentAtNs < $1.sentAtNs }
        )
    }
}
```
