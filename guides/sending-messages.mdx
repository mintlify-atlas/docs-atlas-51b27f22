The XMTP iOS SDK provides flexible methods for sending messages with various content types, delivery options, and error handling strategies.

## Basic Text Messages

### Send Simple Text

The easiest way to send a message:

```swift Conversation.swift:260-269
let conversation = try await client.conversations.list().first!

let messageId = try await conversation.send(text: "Hello, XMTP!")
print("Message sent with ID: \(messageId)")
```

### Send with Generic Content

Use the generic `send()` method for any content type:

```swift Conversation.swift:234-243
let messageId = try await conversation.send(
    content: "Hello, World!"
)
```

## Sending with Options

### Using SendOptions

Configure how your message is sent:

```swift SendOptions.swift:10-22
let options = SendOptions(
    compression: .gzip,  // Optional compression
    contentType: ContentTypeText,  // Explicit content type
    ephemeral: false  // Whether message is ephemeral
)

let messageId = try await conversation.send(
    content: "Compressed message",
    options: options
)
```

### Content Compression

Compress large messages to reduce network usage:

```swift SendOptions.swift:11
// Available compression options
let options = SendOptions(compression: .gzip)

// Send compressed message
try await conversation.send(
    content: longMessage,
    options: options
)
```

## Sending Different Content Types

### Text Messages


### Attachments

Send files, images, and other media:

```swift AttachmentCodec.swift:20-70
// Register the codec
Client.register(codec: AttachmentCodec())

// Prepare attachment
let attachment = Attachment(
    filename: "photo.jpg",
    mimeType: "image/jpeg",
    data: imageData
)

// Send attachment
let messageId = try await conversation.send(
    content: attachment,
    options: SendOptions(contentType: ContentTypeAttachment)
)
```

</Step>

## Advanced Sending Patterns

### Prepare and Publish Pattern

For better control over message sending:

<Steps>

<Step title="Prepare the message">

Prepare without sending to the network:

```swift Group.swift:440-462
let preparedId = try await conversation.prepareMessage(
    content: "Hello!",
    options: nil,
    noSend: false  // false = store and send optimistically
)

print("Message prepared with ID: \(preparedId)")
```

</Step>

<Step title="Publish when ready">

```swift Group.swift:477-483
// Publish a specific prepared message
try await conversation.publishMessage(messageId: preparedId)

// Or publish all prepared messages
try await conversation.publishMessages()
```

</Step>

</Steps>

### Optimistic Sending

Send messages that are stored locally and published asynchronously:

```swift Group.swift:440-462
// Prepare with optimistic sending
let messageId = try await conversation.prepareMessage(
    content: "Sending optimistically",
    noSend: false  // Will send in background
)

// Message is stored locally immediately
// Network publish happens asynchronously
```

### Send with Encoded Content

## Error Handling

Handle common sending errors:

```swift
do {
    let messageId = try await conversation.send(
        content: "Hello!"
    )
    print("Message sent: \(messageId)")
    
} catch CodecError.invalidContent {
    print("Invalid message content")
    
} catch CodecError.codecNotFound {
    print("Content codec not registered")
    
} catch {
    print("Failed to send message: \(error)")
}
```

## Working with Message IDs

Message IDs are returned as hex strings:

```swift
let messageId = try await conversation.send(text: "Hello")

// Store for later reference
UserDefaults.standard.set(messageId, forKey: "lastMessageId")

// Find message by ID later
if let message = try client.conversations.findMessage(messageId: messageId) {
    print("Found message: \(message.content())")
}
```

## Sending in Batch

### Prepare Multiple Messages

```swift
var messageIds: [String] = []

for message in messagesToSend {
```

## Best Practices

<AccordionGroup>
  <Accordion title="Always handle send errors">
    Network issues can cause send failures. Always wrap sends in try-catch:
    
    ```swift
    do {
        let id = try await conversation.send(content: message)
        await updateUI(messageId: id, status: .sent)
    } catch {
        await updateUI(status: .failed, error: error)
        // Optionally retry
    }
    ```
  </Accordion>

  <Accordion title="Use compression for large messages">
    Enable compression for messages over a few KB:
    
    ```swift
    let options = SendOptions(
        compression: message.count > 5000 ? .gzip : nil
    )
    try await conversation.send(content: message, options: options)
    ```
  </Accordion>

  <Accordion title="Store message IDs locally">
    Save message IDs for tracking delivery status:
    
    ```swift
    let messageId = try await conversation.send(content: text)
    
    // Store in database
    database.save(messageId: messageId, status: .sending)
    
    // Update after confirmation
    database.update(messageId: messageId, status: .sent)
    ```
  </Accordion>

  <Accordion title="Use prepare/publish for offline support">
    Prepare messages while offline, publish when connected:
    
    ```swift
    // User composes message (may be offline)
    let messageId = try await conversation.prepareMessage(
        content: message,
        noSend: true
    )
    
    // Store locally
    database.save(messageId: messageId, status: .pending)
    
    // Later, when online
    if isOnline {
        let pendingIds = database.getPendingMessageIds()
        for id in pendingIds {
            try await conversation.publishMessage(messageId: id)
        }
    }
    ```
  </Accordion>

  <Accordion title="Upload large files externally">
    For files over 1MB, use RemoteAttachment:
    
    ```swift
    if fileSize > 1_000_000 {
        // Use remote attachment
        let encrypted = try RemoteAttachment.encodeEncrypted(
            content: attachment,
            codec: AttachmentCodec()
        )
        let url = try await uploadToStorage(encrypted.payload)
        let remote = try RemoteAttachment(url: url, encryptedEncodedContent: encrypted)
        try await conversation.send(content: remote)
    } else {
        // Use direct attachment
        try await conversation.send(content: attachment)
    }
    ```
  </Accordion>

  <Accordion title="Batch message preparation">
    When sending multiple messages, prepare them first:
    
    ```swift
    // Prepare all messages
    for message in messages {
        try await conversation.prepareMessage(
            content: message,
            noSend: true
        )
    }
    
    // Publish all at once (more efficient)
    try await conversation.publishMessages()
    ```
  </Accordion>
</AccordionGroup>

## Examples

### Send Message with Retry

```swift
func sendWithRetry(
    conversation: Conversation,
    content: String,
    maxRetries: Int = 3
) async throws -> String {
    var lastError: Error?
    
    for attempt in 1...maxRetries {
        do {
            let messageId = try await conversation.send(content: content)
            print("Message sent on attempt \(attempt)")
            return messageId
        } catch {
            lastError = error
            print("Attempt \(attempt) failed: \(error)")
            
            if attempt < maxRetries {
                // Wait before retry with exponential backoff
                try await Task.sleep(nanoseconds: UInt64(pow(2.0, Double(attempt))) * 1_000_000_000)
            }
        }
    }
    
    throw lastError!
}

// Usage
let messageId = try await sendWithRetry(
    conversation: conversation,
    content: "Important message"
)
```

### Send with Progress Tracking

```swift
func sendLargeAttachment(
    conversation: Conversation,
    attachment: Attachment,
    progressHandler: @escaping (Double) -> Void
) async throws -> String {
    // 1. Encrypt (20% progress)
    progressHandler(0.2)
    let encrypted = try RemoteAttachment.encodeEncrypted(
        content: attachment,
        codec: AttachmentCodec()
    )
    
    // 2. Upload (20-80% progress)
    progressHandler(0.4)
    let url = try await uploadWithProgress(
        data: encrypted.payload
    ) { uploadProgress in
        progressHandler(0.2 + (uploadProgress * 0.6))
    }
    
    // 3. Send reference (80-100% progress)
    progressHandler(0.9)
    let remote = try RemoteAttachment(
        url: url,
        encryptedEncodedContent: encrypted
    )
    
    let messageId = try await conversation.send(
        content: remote,
        options: SendOptions(contentType: ContentTypeRemoteAttachment)
    )
    
    progressHandler(1.0)
    return messageId
}
```

### Send Rich Reply

```swift
func sendRichReply(
    conversation: Conversation,
    replyingTo message: DecodedMessage,
    replyText: String
) async throws -> String {
    Client.register(codec: ReplyCodec())
    
    let reply = Reply(
        reference: message.id,
        content: replyText,
        contentType: ContentTypeText
    )
    
    return try await conversation.send(
        content: reply,
        options: SendOptions(contentType: ContentTypeReply)
    )
}
```
