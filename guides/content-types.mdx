XMTP uses content type codecs to encode and decode different types of message content. The iOS SDK includes built-in codecs and supports custom implementations.

## Understanding Content Types

Content types define how message data is encoded, decoded, and displayed. Each codec implements the `ContentCodec` protocol:

```swift ContentCodec.swift:80-88
public protocol ContentCodec {
    associatedtype T
    
    var contentType: ContentTypeID { get }
    func encode(content: T) throws -> EncodedContent
    func decode(content: EncodedContent) throws -> T
    func fallback(content: T) throws -> String?
    func shouldPush(content: T) throws -> Bool
}
```

## Built-in Content Types

### Text Messages

The default content type for plain text:

```swift TextCodec.swift:10-52
public let ContentTypeText = ContentTypeID(
    authorityID: "xmtp.org",
    typeID: "text",
    versionMajor: 1,
    versionMinor: 0
)

// Text codec is registered by default
let messageId = try await conversation.send(content: "Hello!")

// Decode text message
let message = messages.first!
let text: String = try message.content()
```

**Properties:**
- Encoding: UTF-8
- Push notifications: Enabled
- Fallback: None (native content)

### Replies

## Creating Custom Content Types

### Define Your Content Type

```swift ContentCodec.swift:80-88
// Define the content type ID
public let ContentTypeCustom = ContentTypeID(
    authorityID: "your-app.example",
    typeID: "custom-type",
    versionMajor: 1,
    versionMinor: 0
)

// Define your data structure
public struct CustomContent: Codable {
    public var title: String
    public var metadata: [String: String]
    public var timestamp: Int64
    
    public init(title: String, metadata: [String: String]) {
        self.title = title
        self.metadata = metadata
        self.timestamp = Date().millisecondsSinceEpoch
    }
}
```

### Implement the Codec

```swift ContentCodec.swift:80-88
public struct CustomCodec: ContentCodec {
    public typealias T = CustomContent
    
    public var contentType = ContentTypeCustom
    
    public init() {}
    
    public func encode(content: CustomContent) throws -> EncodedContent {
        var encodedContent = EncodedContent()
        
        encodedContent.type = ContentTypeCustom
        encodedContent.content = try JSONEncoder().encode(content)
        
        return encodedContent
    }
    
    public func decode(content: EncodedContent) throws -> CustomContent {
        return try JSONDecoder().decode(
            CustomContent.self,
            from: content.content
        )
    }
    
    public func fallback(content: CustomContent) throws -> String? {
        "Custom content: \(content.title)"
    }
    
    public func shouldPush(content: CustomContent) throws -> Bool {
        true
    }
}
```

### Register and Use

```swift
// Register the codec
Client.register(codec: CustomCodec())

// Send custom content
let custom = CustomContent(
    title: "Important Update",
    metadata: ["priority": "high"]
)

let messageId = try await conversation.send(
    content: custom,
    options: SendOptions(contentType: ContentTypeCustom)
)

// Receive custom content
if message.contentTypeId == ContentTypeCustom {
    let custom: CustomContent = try message.content()
    print("Title: \(custom.title)")
}
```

## Working with Content Types

### Check Content Type

```swift
func handleMessage(_ message: DecodedMessage) {
    switch message.contentTypeId {
    case ContentTypeText:
        handleText(message)
    case ContentTypeAttachment:
        handleAttachment(message)
    case ContentTypeReaction:
        handleReaction(message)
    case ContentTypeReply:
        handleReply(message)
    case ContentTypeReadReceipt:
        handleReadReceipt(message)
    default:
        handleUnsupported(message)
    }
}
```

### Use Fallback Content

```swift
func displayMessage(_ message: DecodedMessage) {
    do {
        // Try to decode with the proper codec
        let text: String = try message.content()
        displayText(text)
    } catch {
        // Use fallback for unsupported types
        if let fallback = message.fallback {
            displayFallback(fallback)
        } else {
            displayError("Unsupported message type")
        }
    }
}
```

### Decode Generic Content

```swift ContentCodec.swift:17-31
let encodedContent = message.encodedContent

// Decode using type inference
let text: String = try encodedContent.decoded()
print(text)

// Or decode a specific type
let attachment: Attachment = try encodedContent.decoded()
print(attachment.filename)
```

## Best Practices

<AccordionGroup>
  <Accordion title="Register codecs early">
    Register all codecs at app startup before creating clients:
    
    ```swift
    @main
    struct MyApp: App {
        init() {
            // Register codecs
            Client.register(codec: TextCodec())
            Client.register(codec: AttachmentCodec())
            Client.register(codec: ReactionCodec())
            Client.register(codec: ReplyCodec())
            Client.register(codec: ReadReceiptCodec())
            Client.register(codec: RemoteAttachmentCodec())
        }
    }
    ```
  </Accordion>

  <Accordion title="Always provide fallback text">
    Implement meaningful fallback messages for custom codecs:
    
    ```swift
    public func fallback(content: CustomContent) throws -> String? {
        // Provide context about what the message contains
        "\(content.type): \(content.summary)"
    }
    ```
  </Accordion>

  <Accordion title="Use remote attachments for large files">
    Switch to RemoteAttachment for files over 1MB:
    
    ```swift
    let maxDirectSize = 1_000_000 // 1MB
    
    if fileData.count > maxDirectSize {
        // Use remote attachment
        let encrypted = try RemoteAttachment.encodeEncrypted(
            content: attachment,
            codec: AttachmentCodec()
        )
        let url = try await uploadToStorage(encrypted.payload)
        let remote = try RemoteAttachment(url: url, encryptedEncodedContent: encrypted)
        try await conversation.send(content: remote)
    } else {
        // Use direct attachment
        try await conversation.send(content: attachment)
    }
    ```
  </Accordion>

  <Accordion title="Handle decoding errors gracefully">
    Always handle codec errors and provide fallback UI:
    
    ```swift
    do {
        let content = try message.content() as YourType
        displayContent(content)
    } catch CodecError.invalidContent {
        if let fallback = message.fallback {
            displayFallback(fallback)
        }
    } catch CodecError.codecNotFound {
        displayUnsupported(message.contentTypeId)
    }
    ```
  </Accordion>

  <Accordion title="Version your content types">
    Use semantic versioning for custom content types:
    
    ```swift
    // Version 1.0
    let ContentTypeCustomV1 = ContentTypeID(
        authorityID: "your-app.example",
        typeID: "custom",
        versionMajor: 1,
        versionMinor: 0
    )
    
    // Version 2.0 with breaking changes
    let ContentTypeCustomV2 = ContentTypeID(
        authorityID: "your-app.example",
        typeID: "custom",
        versionMajor: 2,
        versionMinor: 0
    )
    ```
  </Accordion>

  <Accordion title="Disable push for non-critical content">
    Return false from shouldPush for reactions, read receipts, and typing indicators:
    
    ```swift
    public func shouldPush(content: ReadReceipt) throws -> Bool {
        false  // Don't push notifications for read receipts
    }
    ```
  </Accordion>
</AccordionGroup>

## Examples

### Multi-Type Message Handler

```swift
class MessageHandler {
    func handle(_ message: DecodedMessage) async {
        switch message.contentTypeId {
        case ContentTypeText:
            handleText(message)
            
        case ContentTypeAttachment:
            await handleAttachment(message)
            
        case ContentTypeRemoteAttachment:
            await handleRemoteAttachment(message)
            
        case ContentTypeReaction:
            handleReaction(message)
            
        case ContentTypeReply:
            handleReply(message)
            
        case ContentTypeReadReceipt:
            handleReadReceipt(message)
            
        default:
            handleUnsupported(message)
        }
    }
    
    func handleText(_ message: DecodedMessage) {
        let text: String = try! message.content()
        displayTextMessage(text: text, sender: message.senderInboxId)
    }
    
    func handleAttachment(_ message: DecodedMessage) async {
        let attachment: Attachment = try! message.content()
        
        if attachment.mimeType.starts(with: "image/") {
            if let image = UIImage(data: attachment.data) {
                displayImage(image, filename: attachment.filename)
            }
        } else {
            displayFileAttachment(attachment)
        }
    }
    
    func handleRemoteAttachment(_ message: DecodedMessage) async {
        let remote: RemoteAttachment = try! message.content()
        
        showDownloadButton(url: remote.url) {
            let encodedContent = try await remote.content()
            let attachment = try AttachmentCodec().decode(content: encodedContent)
            await self.handleAttachment(attachment)
        }
    }
    
    func handleReaction(_ message: DecodedMessage) {
        let reaction: Reaction = try! message.content()
        updateReactions(messageId: reaction.reference, emoji: reaction.content)
    }
    
    func handleReply(_ message: DecodedMessage) {
        let reply: Reply = try! message.content()
        displayReply(reply, to: reply.reference)
    }
    
    func handleReadReceipt(_ message: DecodedMessage) {
        updateReadStatus(for: message.senderInboxId)
    }
    
    func handleUnsupported(_ message: DecodedMessage) {
        if let fallback = message.fallback {
            displayFallback(fallback)
        } else {
            displayError("Unsupported content type: \(message.contentTypeId)")
        }
    }
}
```

### Smart Attachment Sender

```swift
func sendAttachment(
    conversation: Conversation,
    fileURL: URL,
    uploadService: FileUploadService
) async throws -> String {
    let data = try Data(contentsOf: fileURL)
    let filename = fileURL.lastPathComponent
    let mimeType = getMimeType(for: fileURL)
    
    let attachment = Attachment(
        filename: filename,
        mimeType: mimeType,
        data: data
    )
    
    // Use direct attachment for small files
    if data.count <= 1_000_000 {
        return try await conversation.send(
            content: attachment,
            options: SendOptions(contentType: ContentTypeAttachment)
        )
    }
    
    // Use remote attachment for large files
    let encrypted = try RemoteAttachment.encodeEncrypted(
        content: attachment,
        codec: AttachmentCodec()
    )
    
    let uploadURL = try await uploadService.upload(
        data: encrypted.payload,
        filename: filename
    )
    
    let remote = try RemoteAttachment(
        url: uploadURL,
        encryptedEncodedContent: encrypted
    )
    
    return try await conversation.send(
        content: remote,
        options: SendOptions(contentType: ContentTypeRemoteAttachment)
    )
}
```
