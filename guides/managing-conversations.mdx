The XMTP iOS SDK provides powerful tools for managing both group chats and direct messages through a unified conversations interface.

## Accessing Conversations

Access the conversations manager through your client:

```swift Client.swift:185-188
let client = try await Client.create(
    account: signingKey,
    options: clientOptions
)

let conversations = client.conversations
```

## Listing Conversations

### List All Conversations

Retrieve all conversations (both groups and DMs):

```swift Conversations.swift:47-49
// List all conversations
let allConversations = try await client.conversations.list()

// Iterate through conversations
for conversation in allConversations {
    switch conversation {
    case .group(let group):
        print("Group: \(try group.name())")
    case .dm(let dm):
        print("DM with: \(try dm.peerInboxId)")
    }
}
```

### List Groups Only

```swift Conversations.swift:268-298
let groups = try await client.conversations.listGroups()

for group in groups {
    let name = try group.name()
    let memberCount = try await group.members.count
    print("Group '\(name)' has \(memberCount) members")
}
```

### List Direct Messages Only

```swift
let dms = try await client.conversations.listDms()

for dm in dms {
    let peerInboxId = try dm.peerInboxId
    print("DM with inbox: \(peerInboxId)")
}
```

## Filtering Conversations

### Filter by Creation Time

List conversations created within a time range:

```swift Conversations.swift:268-291
let now = Date().millisecondsSinceEpoch
let oneDayAgo = now - (24 * 60 * 60 * 1000)

let recentGroups = try await client.conversations.listGroups(

)
```

### Filter by Last Activity

Find conversations with recent messages:


## Conversation Settings

### Consent Management

Manage user consent for conversations:

```swift Conversation.swift:82-98
// Get current consent state
let consentState = try conversation.consentState()

switch consentState {
case .allowed:
    print("Conversation is allowed")
case .denied:
    print("Conversation is blocked")
case .unknown:
    print("No consent preference set")
}

// Update consent state
try await conversation.updateConsentState(state: .allowed)
```

### Disappearing Messages

Configure message retention:

```swift Conversation.swift:28-44
// Check if disappearing messages are enabled
let isEnabled = try conversation.isDisappearingMessagesEnabled()

// Get current settings
if let settings = conversation.disappearingMessageSettings {
    print("Messages disappear after \(settings.retentionDurationInNs) ns")
}

// Update settings
let newSettings = DisappearingMessageSettings(
    disappearStartingAtNs: Date().millisecondsSinceEpoch,
    retentionDurationInNs: 24 * 60 * 60 * 1_000_000_000  // 24 hours
)
try await conversation.updateDisappearingMessageSettings(newSettings)

// Clear settings
try await conversation.clearDisappearingMessageSettings()
```

### Filter Conversations by Type

```swift
let allConversations = try await client.conversations.list()

// Filter to only groups
let groups = allConversations.compactMap { conversation -> Group? in
    if case .group(let group) = conversation {
        return group
    }
    return nil
}

// Filter to only DMs

    }
    return nil
```

## Working with Conversation Collections

### Group by Activity

```swift
import Foundation

let conversations = try await client.conversations.list()

let now = Date().millisecondsSinceEpoch
let oneDayAgo = now - (24 * 60 * 60 * 1000)
let oneWeekAgo = now - (7 * 24 * 60 * 60 * 1000)

let today = conversations.filter { $0.lastActivityAtNs > oneDayAgo }
let thisWeek = conversations.filter {
    $0.lastActivityAtNs > oneWeekAgo && $0.lastActivityAtNs <= oneDayAgo
}
let older = conversations.filter { $0.lastActivityAtNs <= oneWeekAgo }

print("Today: \(today.count)")
print("This Week: \(thisWeek.count)")
print("Older: \(older.count)")
```

## Error Handling

```swift
do {
    try await client.conversations.sync()
    let conversations = try await client.conversations.list()
    
    for conversation in conversations {
        let members = try await conversation.members()
        print("Conversation has \(members.count) members")
    }
} catch ConversationError.memberNotRegistered(let addresses) {
    print("Members not registered: \(addresses)")
} catch ConversationError.streamingFailure {
    print("Failed to stream conversations")
} catch {
    print("Unexpected error: \(error)")
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Sync before listing">
    Always sync conversations before listing to ensure you have the latest data:
    
    ```swift
    try await client.conversations.sync()
    let conversations = try await client.conversations.list()
    ```
  </Accordion>

  <Accordion title="Use appropriate filters">
    Apply filters to reduce data transfer and improve performance:
    
    ```swift
    // Only fetch recent, allowed conversations
    let conversations = try await client.conversations.listGroups(
        lastActivityAfterNs: recentTimestamp,
        consentStates: [.allowed],
        limit: 50
    )
    ```
  </Accordion>

  <Accordion title="Cache conversation references">
    Store conversation IDs rather than full conversation objects:
    
    ```swift
    let conversationIds = conversations.map { $0.id }
    UserDefaults.standard.set(conversationIds, forKey: "recentConversations")
    
    // Later, retrieve by ID
    if let conversation = try await client.conversations.findConversation(
        conversationId: storedId
    ) {
        // Use conversation
    }
    ```
  </Accordion>

  <Accordion title="Handle pagination">
    Use limits and ordering for efficient pagination:
    
    ```swift
    var allConversations: [Group] = []
    let pageSize = 50
    var lastTimestamp: Int64? = nil
    
    repeat {
        let page = try await client.conversations.listGroups(
            lastActivityBeforeNs: lastTimestamp,
            limit: pageSize,
            orderBy: .lastActivity
        )
        
        allConversations.append(contentsOf: page)
        lastTimestamp = page.last?.lastActivityAtNs
        
    } while lastTimestamp != nil
    ```
  </Accordion>
</AccordionGroup>
